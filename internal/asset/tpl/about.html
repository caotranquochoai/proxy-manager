Proxy manager.

Support http、https、socks4、socks4a、socks5、shadowsocks proxy.

Project : https://github.com/hidu/proxy-manager

Current Version: {{.version}}




Usage:

1. As Proxy Server

not support https url. 
ch = curl_init(); 
curl_setopt($ch, CURLOPT_URL, "https://hidu.github.io/hello.md"); 
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_PROXY,"{{.proxy_host}}:{{.proxy_port}}");
curl_setopt($ch, CURLOPT_PROXYUSERPWD,"name:psw");// if necessary
$body=curl_exec($ch); 
curl_close($ch);

curl -x http://$name:$psw@{{.proxy_host}}:{{.proxy_port}} 'http://hidu.github.io/hello.md'

2. As Gateway Server

support http、https url. 
curl 'http://$name:$psw@{{.proxy_host}}:{{.proxy_port}}/query?url=https://hidu.github.io/hello.md
  
curl 'http://$name:$psw@{{.proxy_host}}:{{.proxy_port}}/query?method=POST&url=https://hidu.github.io/hello.md&headers={"a":["a"]}' \
  -X POST --data "request body"
